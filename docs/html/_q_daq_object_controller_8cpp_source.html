62 bool isQDaqType(int type) {
63  return type == qMetaTypeId<QDaqObject*>() ||
64  type == qMetaTypeId<QDaqObjectList>() ||
65  type == qMetaTypeId<QDaqVector>() ||

68 
69  return false;

72 bool isQDaqType(const QVariant& v) {
73  int type = v.userType();
74 
75  if (type == qMetaTypeId<QDaqObject*>() ||
76  type == qMetaTypeId<QDaqObjectList>() ||
77  type == qMetaTypeId<QDaqVector>() ||
78  type == qMetaTypeId<QStringList>() ||
79  type == qMetaTypeId<QVariantList>()) return true;
80 
81  if (type==QMetaType::QObjectStar) {
82  QObject* obj = v.value<QObject*>();
83  QDaqObject* dacobj = qobject_cast<QDaqObject*>(obj);
84  if (dacobj) return true;
85  }
86 
87  return false;
88 }

90 template <class T>
91 QString qdaqVectorTypeToString(const QVariant& var)
92 {
93  T v = var.value<T>();
94  QString str('[');
95  for(int i=0; i<v.size(); i++)
96  {
97  str += QString::number(v[i]);
98  if (i<v.size()-1) str += ',';
99  }
100  str += ']';
101  return str;
102 }
103 QString qdaqTypeToString(const QVariant& v)
104 {
105  int type = v.userType();
106  if (type == qMetaTypeId<QDaqVector>()) return qdaqVectorTypeToString<QDaqVector>(v);
107  else if (type == QMetaType::QObjectStar) {
108  QDaqObject* obj = qobject_cast<QDaqObject*>(v.value<QObject*>());
109  if (obj) return obj->path();
110  else return QString();
111  }
112  else if (type == qMetaTypeId<QDaqObject*>()) {
113  QDaqObject* obj = v.value<QDaqObject*>();
114  if (obj) return obj->path();
115  else return QString();
116  }
117  else if (type == qMetaTypeId<QDaqObjectList>()) {
118  QDaqObjectList lst = v.value<QDaqObjectList>();
119  QString str('[');
120  for(int i=0; i<lst.size(); i++) {
121  QDaqObject* obj = lst.at(i);
122  if (obj) str += obj->path();
123  else str += "0x0";
124  if (i<lst.size()-1) str += ",";
125  }
126  str += "]";
127  return str;
128  }
129  else if (type == QMetaType::QStringList) {
130  QStringList lst = v.value<QStringList>();
131  QString str('[');
132  for(int i=0; i<lst.size(); i++) {
133  QString s = lst.at(i);
134  str += s;
135  if (i<lst.size()-1) str += ",";
136  }
137  str += "]";
138  return str;
139  }
140  else if (type == QMetaType::QVariantList) {
141  QVariantList lst = v.value<QVariantList>();
142  QString str('[');
143  for(int i=0; i<lst.size(); i++) {
144  QString s = lst.at(i).toString();
145  str += s;
146  if (i<lst.size()-1) str += ",";
147  }
148  str += "]";
149  return str;
150  }
151  else return QString();
152 }

253 int QDaqObjectControllerPrivate::enumToInt(const QMetaEnum &metaEnum, int enumValue) const
254 {
255  QMap<int, int> valueMap; // dont show multiple enum values which have the same values
256  int pos = 0;
257  for (int i = 0; i < metaEnum.keyCount(); i++) {
258  int value = metaEnum.value(i);
259  if (!valueMap.contains(value)) {
260  if (value == enumValue)
261  return pos;
262  valueMap[value] = pos++;
263  }
264  }
265  return -1;

284 bool QDaqObjectControllerPrivate::isSubValue(int value, int subValue) const
285 {
286  if (value == subValue)
287  return true;
288  int i = 0;
289  while (subValue) {
290  if (!(value & (1 << i))) {
291  if (subValue & 1)
292  return false;
293  }
294  i++;
295  subValue = subValue >> 1;
296  }
297  return true;
298 }

311 int QDaqObjectControllerPrivate::flagToInt(const QMetaEnum &metaEnum, int flagValue) const
312 {
313  if (!flagValue)
314  return 0;
315  int intValue = 0;
316  QMap<int, int> valueMap; // dont show multiple enum values which have the same values
317  int pos = 0;
318  for (int i = 0; i < metaEnum.keyCount(); i++) {
319  int value = metaEnum.value(i);
320  if (!valueMap.contains(value) && isPowerOf2(value)) {
321  if (isSubValue(flagValue, value))
322  intValue |= (1 << pos);
323  valueMap[value] = pos++;
324  }
325  }
326  return intValue;
327 }

329 int QDaqObjectControllerPrivate::intToFlag(const QMetaEnum &metaEnum, int intValue) const
330 {
331  QMap<int, bool> valueMap; // dont show multiple enum values which have the same values
332  QList<int> values;
333  for (int i = 0; i < metaEnum.keyCount(); i++) {
334  int value = metaEnum.value(i);
335  if (!valueMap.contains(value) && isPowerOf2(value)) {
336  valueMap[value] = true;
337  values.append(value);
338  }
339  }
340  int flagValue = 0;
341  int temp = intValue;
342  int i = 0;
343  while (temp) {
344  if (i >= values.count())
345  return -1;
346  if (temp & 1)
347  flagValue |= values.at(i);
348  i++;
349  temp = temp >> 1;
350  }
351  return flagValue;
352 }

354 void QDaqObjectControllerPrivate::updateClassProperties(const QMetaObject *metaObject, bool recursive)
355 {
356  if (!metaObject)
357  return;
358 
359  if (recursive)
360  updateClassProperties(metaObject->superClass(), recursive);
361 
362  QtProperty *classProperty = m_classToProperty.value(metaObject);
363  if (!classProperty)
364  return;
365 
366  for (int idx = metaObject->propertyOffset(); idx < metaObject->propertyCount(); idx++) {
367  QMetaProperty metaProperty = metaObject->property(idx);
368  if (metaProperty.isReadable()) {
369  if (m_classToIndexToProperty.contains(metaObject) && m_classToIndexToProperty[metaObject].contains(idx)) {
370  QtVariantProperty *subProperty = m_classToIndexToProperty[metaObject][idx];
371  if (metaProperty.isEnumType()) {
372  if (metaProperty.isFlagType())
373  subProperty->setValue(flagToInt(metaProperty.enumerator(), metaProperty.read(m_object).toInt()));
374  else
375  {
376  int i = *reinterpret_cast<const int *>(metaProperty.read(m_object).constData());
377  subProperty->setValue(enumToInt(metaProperty.enumerator(), i));
378  }
379  } else {
380  QVariant v = metaProperty.read(m_object);
381  if (isQDaqType(v)) subProperty->setValue(qdaqTypeToString(v));
382  else subProperty->setValue(v);
383  }
384  }
385  }
386  }

508 void QDaqObjectControllerPrivate::addDynProperties()
509 {
510  QtProperty *classProperty = m_classToProperty.value(&QDaqObject::staticMetaObject);
511 
512  QList<QtVariantProperty*> lst = m_dynPropNameToProperty.values();
513  QListIterator<QtVariantProperty *> it(lst);
514  while (it.hasNext()) {
515  classProperty->removeSubProperty(it.next());
516  }
517  m_dynPropNameToProperty.clear();
518  m_propertyToDynPropName.clear();
519 
520  QByteArrayList dynProps = m_object->dynamicPropertyNames();
521  QStringList sorted;
522  foreach(const QByteArray& propName, dynProps) sorted << QString(propName);
523  sorted.sort();
524 
525  foreach(const QString& propNameStr, sorted) {
526  QByteArray propName = propNameStr.toLatin1();
527  QVariant var = m_object->property(propName.constData());
528  int type = var.userType();
529  QtVariantProperty *subProperty = 0;
530  if (isQDaqType(var)) {
531  subProperty = m_readOnlyManager->addProperty(QVariant::String, QLatin1String(propName));
532  subProperty->setValue(qdaqTypeToString(var));
533  subProperty->setEnabled(false);
534  } else if (m_manager->isPropertyTypeSupported(type)) {
535  subProperty = m_manager->addProperty(type, QLatin1String(propName));
536  subProperty->setValue(var);
537  } else if (type == QVariant::UInt) {
538  type = QVariant::Int;
539  subProperty = m_manager->addProperty(type, QLatin1String(propName));
540  subProperty->setValue(var);
541  subProperty->setAttribute("minimum",QVariant((int)0));
542  } else { // could not convert the type
543  subProperty = m_readOnlyManager->addProperty(QVariant::String, QLatin1String(propName));
544  QString str = var.toString();
545  if (str.isEmpty()) {
546  subProperty->setValue(QLatin1String(var.typeName()));
547  subProperty->setEnabled(false);
548  } else subProperty->setValue(str);
549 
550  }
551  classProperty->addSubProperty(subProperty);
552  m_dynPropNameToProperty[propName] = subProperty;
553  m_propertyToDynPropName[subProperty] = propName;
554  }
555 }

568 {

570 
571 
572  int idx = m_propertyToIndex.value(property);
573 
574  const QMetaObject *metaObject = m_object->metaObject();

576  if (metaProperty.isEnumType()) {

579  else

583  }

592  QByteArray propName = m_propertyToDynPropName.value((QtVariantProperty*)property);
593  m_object->setProperty(propName,value);
594  updateDynProperties();
595  }
596 }
