127 class TimeScaleEngine: public QwtLinearScaleEngine
128 {
129 protected:
130  static double conversion_factor(double T)
131  {
132  double C = 1.; // conversion const
133  if (T<=60) C = 1.; // secs up to 1 min
134  else if (T<=3600) C = 60; // mins up to 1hr
135  else if (T<=24*3600) C = 60*60; // hr up to 1 day
136  else C = 24*60*60; // else in days
137  return C;
138  }
139 public:
140  virtual void autoScale(int maxSteps,
141  double &x1, double &x2, double &stepSize) const
142  {
143  double C = conversion_factor(fabs(x2-x1));
144  x1 /= C; x2 /= C; // convert to s,m,hr,d ...
145  QwtLinearScaleEngine::autoScale(maxSteps,x1,x2,stepSize);
146  x1 *= C; x2 *= C; stepSize *= C; // convert to s
147  }
148 };

378 void QDaqPlotWidget::plot(const QDaqVector &x, const QDaqVector &y, const QString &attr, const QColor &clr)
379 {
380  static const Qt::GlobalColor eight_colors[8] =
381  {
382  Qt::blue,
383  Qt::red,
384  Qt::darkGreen,
385  Qt::magenta,
386  Qt::darkBlue,
387  Qt::darkMagenta,
388  Qt::darkCyan,
389  Qt::darkRed
390  };
391 
392  //MATLAB-Octave-style linestyles for reference
393  //static const QString linestyles[5] =
394  //{"-","--",":",".-","none"};
395 
396  //Respective Qwt names for QwtPlotCurve::setPen Style: (a Qt::PenStyle object)
397  static const Qt::PenStyle penstyles[4] =
398  {Qt::SolidLine,Qt::DashLine,Qt::DotLine,Qt::DashDotLine};
399 
400  //MATLAB-Octave-style markerstyles for reference (not all)
401  //static const QString markerstyles[12] =
402  //{"+","o","*",".","x","s","d","^","v",">","<","h"};
403 
404  //Respective Qwt names for QwtSymbol::Style
405  static const QwtSymbol::Style dotstyles[12] =
406  {QwtSymbol::Cross,QwtSymbol::Ellipse,QwtSymbol::Star1,QwtSymbol::NoSymbol,QwtSymbol::XCross,
407  QwtSymbol::Rect,QwtSymbol::Diamond,QwtSymbol::UTriangle,QwtSymbol::DTriangle,QwtSymbol::RTriangle,
408  QwtSymbol::LTriangle,QwtSymbol::Hexagon};
409 
410  QwtPlotCurve* curve = new QwtPlotCurve;
411  curve->setData(new QDaqPlotData(x,y));
412 
413  QColor plotclr = (clr.isValid()) ? clr : QColor(eight_colors[id_++ & 0x07]);
414 
415  QString lines, markers;
416 
417  QString pattern("(?<line>(-|--|:|.-|none))(?<marker>(\\+|o|\\*|\\.|x|s|d|\\^|v|>|<|h))");
418  //use the pattern as a regular expression
419  QRegularExpression re(pattern);
420  //try to match the plot attributes passed by the user to the reg expr
421  QRegularExpressionMatch match = re.match(attr,0);
422 
423  QwtSymbol * symbol = new QwtSymbol;
424  //set a discernible but not very intrusive size for markers
425  symbol->setSize(4,4);
426  //make the marker same color as plot
427  symbol->setColor(plotclr);
428 
429  //if we find a match in user input and pattern, separate the two fields of match
430  if (match.hasMatch()) {
431  lines = match.captured("line");
432  markers = match.captured("marker");
433  }
434  else
435  {
436  //error message, instructions for proper usage, select default plot !!!!
437 // qInfo ("Please give linestyle/markerstyle attributes, following MATLAB style");
438 // qInfo ("I will now select some default: cont. lines, no marker");
439  curve->setStyle(QwtPlotCurve::Lines);
440  curve->setSymbol(NULL);
441  }
442 
443 
444  //Ok, now select linestyle
445  if (lines == "none")
446  {
447  //if user desires no line, at least plot markers, instead of nothing!
448  curve->setStyle(QwtPlotCurve::Dots);
449  }
450  else if (lines == "-")
451  {
452  curve->setStyle(QwtPlotCurve::Lines);
453  curve->setPen(plotclr, 0.0, penstyles[0]);
454  }
455  else if (lines == "--")
456  {
457  curve->setStyle(QwtPlotCurve::Lines);
458  curve->setPen(plotclr, 0.0, penstyles[1]);
459  }
460  else if (lines == ":")
461  {
462  curve->setStyle(QwtPlotCurve::Lines);
463  curve->setPen(plotclr, 0.0, penstyles[2]);
464  }
465  else if (lines == ".-")
466  {
467  curve->setStyle(QwtPlotCurve::Lines);
468  curve->setPen(plotclr, 0.0, penstyles[3]);
469  }
470  else
471  {
472  curve->setStyle(QwtPlotCurve::Lines);
473  curve->setPen(plotclr, 0.0, penstyles[0]);
474  }
475 
476 //Now on for the marker style...
477  if (markers == ".")
478  {
479  //if user desires no marker, indulge them: plot only a line
480  curve->setStyle(QwtPlotCurve::Lines);
481  curve->setSymbol(NULL);
482  }
483  else if (markers == "+")
484  {
485  symbol->setStyle(dotstyles[0]);
486  }
487  else if (markers == "o")
488  {
489  symbol->setStyle(dotstyles[1]);
490  }
491  else if (markers == "*")
492  {
493  symbol->setStyle(dotstyles[2]);
494  }
495  else if (markers == "x")
496  {
497  symbol->setStyle(dotstyles[4]);
498  }
499  else if (markers == "s")
500  {
501  symbol->setStyle(dotstyles[5]);
502  }
503  else if (markers == "d")
504  {
505  symbol->setStyle(dotstyles[6]);
506  }
507  else if (markers == "^")
508  {
509  symbol->setStyle(dotstyles[7]);
510  }
511  else if (markers == "v")
512  {
513  symbol->setStyle(dotstyles[8]);
514  }
515  else if (markers == ">")
516  {
517  symbol->setStyle(dotstyles[9]);
518  }
519  else if (markers == "<")
520  {
521  symbol->setStyle(dotstyles[10]);
522  }
523  else if (markers == "h")
524  {
525  symbol->setStyle(dotstyles[11]);
526  }
527  else{
528  symbol->setStyle(dotstyles[3]);
529  }
530  curve->setSymbol(symbol);
531 
532  curve->attach(this);
533 
534  replot();
