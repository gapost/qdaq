129 QStringList QDaqConsole::introspection(const QString& lookup)
130 {
131  // list of found tokens
132  QStringList properties, children, functions;
133 
134  if (lookup.isEmpty()) return properties;
135 
136  QScriptEngine* eng = session->getEngine();
137  QScriptValue scriptObj = eng->evaluate(lookup);
138 
139  // if the engine cannot recognize the variable return
140  if (eng->hasUncaughtException()) return properties;
141 
142  // if a QObject add the named children
143  if (scriptObj.isQObject())
144  {
145  QObject* obj = scriptObj.toQObject();
146 
147  foreach(QObject* ch, obj->children())
148  {
149  QString name = ch->objectName();
150  if (!name.isEmpty())
151  children << name;
152  }
153 
154  }
155 
156  // add the script properties
157  {
158  QScriptValue obj(scriptObj); // the object to iterate over
159  while (obj.isObject()) {
160  QScriptValueIterator it(obj);
161  while (it.hasNext()) {
162  it.next();
163 
164  // avoid array indices
165  bool isIdx;
166  it.scriptName().toArrayIndex(&isIdx);
167  if (isIdx) continue;
168 
169  // avoid "hidden" properties starting with "__"
170  if (it.name().startsWith("__")) continue;
171 
172  // include in list
173  if (it.value().isQObject()) children << it.name();
174  else if (it.value().isFunction()) functions << it.name();
175  else properties << it.name();
176  }
177  obj = obj.prototype();
178  }
179  }
180 
181  children.removeDuplicates();
182  children.sort(Qt::CaseInsensitive);
183  functions.removeDuplicates();
184  functions.sort(Qt::CaseInsensitive);
185  properties.removeDuplicates();
186  properties.sort(Qt::CaseInsensitive);
187 
188  children.append(properties);
189  children.append(functions);
190 
191  return children;
192 
193 }
