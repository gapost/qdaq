366 bool QDaqObject::setQDaqProperty(QString name, const QScriptValue &value)
367 {
368  // Check the name
369  static const char* errmsg[] = {
370  "Empty string is not allowed for property name.",
371  "Property names must start with a letter.",
372  "Property names may only contain letters, numbers and the underscore."
373  };
374  int cd;
375  if (!isNameValid(name,&cd))
376  {
377  throwScriptError(errmsg[cd]);
378  return false;
379  }
380  if (!isNameUnique(name))
381  {
382  throwScriptError("Name is used by a child object.");
383  return false;
384  }
385  const QMetaObject* obj = metaObject();
386  QByteArray nm = name.toLatin1();
387  if (obj->indexOfConstructor(nm)+1 ||
388  obj->indexOfEnumerator(nm)+1 ||
389  obj->indexOfMethod(nm)+1 ||
390  obj->indexOfProperty(nm)+1 ||
391  obj->indexOfSignal(nm)+1 ||
392  obj->indexOfSlot(nm)+1) {
393  throwScriptError("New property name masks QDaq object properties.");
394  return false;
395  }
396 
397  // Do some QDaq specific type conversions
398  QVariant V;
399  bool ok = false;
400 
401  // check for QDaq JS classes Vector and ByteArray
402  QScriptEngine* eng = engine();
403  if (eng) {
404  QScriptValue ByteArray = eng->globalObject().property("ByteArray");
405  if (ByteArray.isValid() && value.instanceOf(ByteArray)) {
406  QByteArray ba = qscriptvalue_cast<QByteArray>(value);
407  V = QVariant::fromValue(ba);
408  ok = true;
409  }
410  if (!ok) {
411  QScriptValue Vector = eng->globalObject().property("Vector");
412  if (Vector.isValid() && value.instanceOf(Vector)) {
413  QDaqVector v = qscriptvalue_cast<QDaqVector>(value);
414  V = QVariant::fromValue(v);
415  ok = true;
416  }
417  }
418  }
419 
420  // check for QDaqObject
421  if (!ok && value.isQObject()) {
422  QDaqObject* obj(0);
423  QDaqTypes::fromScriptValue(value,obj);
424  if (obj) {
425  V = QVariant::fromValue(obj);
426  ok = true;
427  }
428  }
429 
430  // check for QDaqObject list
431  if (!ok && value.isArray()) {
432  quint32 len = value.property("length").toUInt32();
433  bool isObjLst = true;
434  QDaqObjectList L;
435  for (quint32 i = 0; i < len; ++i) {
436  QScriptValue p = value.property(i);
437  if (!p.isQObject()) { isObjLst=false; break; }
438  QDaqObject* obj(0);
439  QDaqTypes::fromScriptValue(p,obj);
440  if (obj==0) { isObjLst=false; break; }
441  else L << obj;
442  }
443  if (isObjLst) {
444  V = QVariant::fromValue(L);
445  ok = true;
446  }
447  }
448 
449 
450  // if all failed do the default Qt conversion
451  if (!ok) V = value.toVariant();
452 
453  setProperty(name.toLatin1(),V);
454  return true;
455 }
