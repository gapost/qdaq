136 bool QDaqChannel::average()
137 {
138  int m = (counter_ < depth_) ? counter_ : depth_;
139  if (m==0) return false;
140 
141  if (type_==None || m==1)
142  {
143  v_ = buff_[0];
144  dv_ = 0;
145  return true;
146  }
147 
148  v_ = dv_ = 0;
149 
150  double wt;
151  int sw, m2;
152  switch(type_)
153  {
154  case Running:
155  for(int i=0; i<m; ++i)
156  {
157  double y = buff_[i];
158  v_ += y;
159  dv_ += y*y;
160  }
161  v_ /= m;
162  dv_ /= m;
163  break;
164  case Median:
165  for(int i=0; i<m; ++i)
166  {
167  sorted_buffer[i] = buff_[i];
168  double y = buff_[i];
169  dv_ += y*y;
170  }
171  std::sort(sorted_buffer.begin(),sorted_buffer.end()); //needs #include <algorithm>
172  m2 = m/2;
173  if (m % 2) { v_ = sorted_buffer[m2];}
174  else v_ = (sorted_buffer[m2-1] + sorted_buffer[m2])/2.0;
175  dv_ /= m;
176  break;
177  case Delta:
178  sw = ((counter_ & 1) == 1) ? -1 : 1; // get the current sign
179  for(int i=1; i<m; ++i)
180  {
181  double y = (buff_[i]-buff_[i-1]);
182  v_ += y*sw;
183  dv_ += y*y;
184  sw = - sw;
185  }
186  v_ /= (2*(m-1));
187  dv_ /= (4*(m-1));
188  break;
189  case ForgettingFactor:
190  wt = 1-ff_; // weight factor
191  for(int i=0; i<m; ++i)
192  {
193  double y = buff_[i];
194  v_ += y*wt;
195  dv_ += y*y*wt;
196  wt *= ff_;
197  }
198  v_ *= ffw_;
199  dv_ *= ffw_;
200  break;
201  case None:
202  break;
203  }
204 
205  // dv now contains <y^2>
206  // convert to st. deviation of <y> = sqrt(<y^2>-<y>^2)
207  dv_ -= v_*v_;
208  if (dv_>0) dv_=sqrt(dv_);
209  else dv_ = 0;
210 
211  return true;
212 }

214 bool QDaqChannel::run()
215 {
216  if (!QDaqJob::run()) return false;
217 
218  dataReady_ = true;
219  switch (channeltype_)
220  {
221  case Clock:
222  push(QDaqTimeValue::now());
223  break;
224  case Random:
225  push(1.*rand()/RAND_MAX);
226  break;
227  case Inc:
228  push(v_ + 1);
229  break;
230  case Dec:
231  push(v_ - 1);
232  break;
233  case Normal:
234  default:
235  break;
236  }
237 
238  if ((dataReady_=average()))
239  {
240  if (parser_)
241  {
242  try
243  {
244  v_ = parser_->Eval();
245  }
246  catch (mu::Parser::exception_type &e)
247  {
248  //Q_UNUSED(e);
249 
250  const mu::string_type& msg = e.GetMsg();
251 #if defined(_UNICODE)
252  pushError(QString("muParser"),QString::fromStdWString(msg));
253 #else
254  pushError(QString("muParser"),QString(msg.c_str()));
255 #endif
256  //std::cout << e.GetMsg() << endl;
257  dataReady_ = false;
258  }
259  }
260 
261  {
262  v_ = multiplier_*v_ + offset_;
263  dv_ = multiplier_*dv_ + offset_;
264  }
265 
266  // handle over/under flow
267  if (v_ < range_[0]) v_ = range_[0];
268  if (v_ > range_[1]) v_ = range_[1];
269 
270  dataReady_ = std::isfinite(v_);
271 
272  emit updateWidgets();
273  }
274 
275  return true;
